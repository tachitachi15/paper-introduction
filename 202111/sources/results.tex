\begin{itemize}
    \item ターゲット　($R_T,v_T, \theta_T$) = ($100m, 20m/s, 1$),($80m, 5m/s, 10$)
    \item 干渉($R_I,v_I, \theta_I$) = (15m, -15m/s, -3)
    \item 送信信号と干渉信号のチャープの傾きは異なる(fig2(b)のケース)

\end{itemize}

\end{itemize}
